const container = document.getElementById('brain-viewport'); const scene = new THREE.Scene(); scene.fog = new THREE.FogExp2(0x0a192f, 0.12); const camera = new THREE.PerspectiveCamera(35, container.clientWidth / container.clientHeight, 0.1, 1000); camera.position.z = 8; const renderer = new THREE.WebGLRenderer({ antialias: true, alpha: true }); renderer.setSize(container.clientWidth, container.clientHeight); renderer.setPixelRatio(window.devicePixelRatio); container.appendChild(renderer.domElement); scene.add(new THREE.AmbientLight(0xffffff, 0.3)); const pLight = new THREE.PointLight(0xffffff, 1); pLight.position.set(5, 5, 10); scene.add(pLight); // --- SISTEMA DE PARTÍCULAS --- const partGeo = new THREE.BufferGeometry(); const partCount = 1000; const posArr = new Float32Array(partCount * 3); for(let i=0; i < partCount * 3; i++) posArr[i] = (Math.random() - 0.5) * 15; partGeo.setAttribute('position', new THREE.BufferAttribute(posArr, 3)); const partMat = new THREE.PointsMaterial({ size: 0.05, color: 0xffffff, transparent: true, opacity: 0.5, blending: THREE.AdditiveBlending }); const particles = new THREE.Points(partGeo, partMat); scene.add(particles); let brain = null; let tX = 0, tY = 0; let modoAtual = 'NORMAL'; let pulseSpeed = 0.002; let erroAtivo = false; // Variável para o ruído de erro const loader = new THREE.GLTFLoader(); loader.load('human-brain.glb', (gltf) => { brain = gltf.scene; brain.traverse(c => { if (c.isMesh) { c.material = new THREE.MeshStandardMaterial({ color: 0x110000, emissive: 0xe30613, emissiveIntensity: 0.8, roughness: 0.3, metalness: 0.5 }); } }); scene.add(brain); }); // FUNÇÃO DE SIMULAÇÃO CLÍNICA function simular(modo) { modoAtual = modo; const label = document.getElementById('status-label'); if(modo === 'TDAH') { pulseSpeed = 0.008; particles.material.color.setHex(0xffaa00); label.innerText = "STATUS: INSTABILIDADE DOPAMINÉRGICA"; } else if (modo === 'TEA') { pulseSpeed = 0.0008; particles.material.color.setHex(0x0088ff); label.innerText = "STATUS: HIPERCONECTIVIDADE LOCAL"; } else { pulseSpeed = 0.002; particles.material.color.setHex(0xffffff); label.innerText = "STATUS: PADRÃO ESTÁVEL"; } } // VALIDAÇÃO COM RUÍDO DE ERRO function validarAcesso() { const val = document.getElementById('idInput').value.toUpperCase(); if (val === "MATH001" || val.startsWith("PAC")) { window.location.href = "unt.html"; } else { // Dispara o Ruído erroAtivo = true; if(brain) { brain.traverse(c => { if (c.isMesh) { c.material.emissive.setHex(0xff0000); c.material.emissiveIntensity = 10; } }); } setTimeout(() => { erroAtivo = false; if(brain) brain.traverse(c => { if (c.isMesh) c.material.emissive.setHex(0xe30613); }); }, 400); } } window.addEventListener('mousemove', (e) => { tX = (e.clientX / window.innerWidth - 0.5) * 0.2; tY = (e.clientY / window.innerHeight - 0.5) * 0.2; }); function animate() { requestAnimationFrame(animate); particles.rotation.y += (modoAtual === 'TDAH' ? 0.005 : 0.0007); if (brain) { brain.rotation.y += 0.004; brain.rotation.x += 0.05 * (tY - brain.rotation.x); brain.rotation.z += 0.05 * (tX - brain.rotation.z); // --- LÓGICA DE RUÍDO E CHACOALHO --- if (erroAtivo) { // Chacoalho violento de erro (Ruído de Sistema) brain.position.x = (Math.random() - 0.5) * 0.4; brain.position.y = (Math.random() - 0.5) * 0.4; particles.scale.set(1.5, 1.5, 1.5); } else if (modoAtual === 'TDAH') { // Jitter sutil do TDAH (Ruído Neural) brain.position.x = (Math.random() - 0.5) * 0.05; brain.position.y = (Math.random() - 0.5) * 0.05; particles.scale.set(1, 1, 1); } else { brain.position.set(0, 0, 0); particles.scale.set(1, 1, 1); } // Pulsação let p = 0.7 + Math.sin(Date.now() * pulseSpeed) * 0.3; if(modoAtual === 'TDAH') p += (Math.random() - 0.5) * 0.15; brain.traverse(c => { if (c.isMesh && !erroAtivo) { if (c.material.emissiveIntensity < 1.5) { c.material.emissiveIntensity = p; if(modoAtual === 'TDAH') c.material.emissive.setHex(0xffaa00); else if(modoAtual === 'TEA') c.material.emissive.setHex(0x0088ff); else c.material.emissive.setHex(0xe30613); } else { c.material.emissiveIntensity *= 0.95; } } }); } renderer.render(scene, camera); } animate(); window.addEventListener('resize', () => { camera.aspect = container.clientWidth / container.clientHeight; camera.updateProjectionMatrix(); renderer.setSize(container.clientWidth, container.clientHeight); });
